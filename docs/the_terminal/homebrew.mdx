import { useState, useEffect } from 'react';

export const ImageLoop = ({ images }) => {
  const [index, setIndex] = useState(0);
  const [isZoomed, setIsZoomed] = useState(false);
  
  useEffect(() => {
    if (images.length <= 1) return;
    const timer = setInterval(() => {
      setIndex((prevIndex) => (prevIndex + 1) % images.length);
    }, 2000);
    return () => clearInterval(timer);
  }, [images.length]);

  return (
    <div style={{ marginBottom: '20px' }}>
      <div 
        onClick={() => setIsZoomed(!isZoomed)}
        style={{ 
          cursor: 'zoom-in', 
          position: 'relative',
          borderRadius: '12px',
          overflow: 'hidden',
          border: '2px solid var(--ifm-color-emphasis-300)',
          boxShadow: '0 8px 24px rgba(0,0,0,0.15)'
        }}
      >
        <img src={images[index]} style={{ display: 'block', width: '100%' }} />
        
        {images.length > 1 && (
          <div style={{
            position: 'absolute', bottom: '10px', right: '10px',
            backgroundColor: '#333', color: 'white',
            padding: '2px 12px', borderRadius: '20px', fontSize: '0.75rem', 
            fontWeight: 'bold', opacity: '0.8'
          }}>
            Step {index + 1} of {images.length}
          </div>
        )}
      </div>

      {isZoomed && (
        <div 
          onClick={() => setIsZoomed(false)}
          style={{
            position: 'fixed', top: 0, left: 0, width: '100%', height: '100%',
            backgroundColor: 'rgba(0,0,0,0.9)', zIndex: 9999,
            display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'zoom-out'
          }}
        >
          <img src={images[index]} style={{ maxHeight: '95vh', maxWidth: '95vw', borderRadius: '8px' }} />
        </div>
      )}
    </div>
  );
};

# Homebrew

**Homebrew** ဆိုသည်မှာ macOS နှင့် Linux တို့အတွက် အခမဲ့အသုံးပြုနိုင်သော open-source software package manager တစ်ခု ဖြစ်ပါသည်။ Terminal ပေါ်တွင် command-line tools များ၊ developer dependencies များနှင့် applications များကို အလွယ်တကူ install လုပ်ခြင်း၊ update ပြုလုပ်ခြင်းနှင့် စီမံခန့်ခွဲခြင်းတို့အတွက် အသုံးပြုရပါသည်။ ၎င်းသည် terminal အတွက် "App Store" တစ်ခုကဲ့သို့ အလုပ်လုပ်ပေးပြီး software များကို သီးခြား directory တစ်ခုတွင် စနစ်တကျ install လုပ်ပေးပါသည်။

* 🔗 [Official Software Website](https://brew.sh)

## Installation Script
Homebrew ကို ထည့်သွင်းရန် အောက်ပါ script ကို copy ကူးယူပြီး Terminal တွင် paste ပြုလုပ်၍ run ပေးပါ။

```bash
/bin/bash -c "$(curl -fsSL [https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh](https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh))"
```

## Maintenance Commands

Homebrew ကို ကောင်းမွန်စွာ အလုပ်လုပ်စေရန် အောက်ပါ command များဖြင့် ထိန်းသိမ်းနိုင်ပါသည်။
နားလည်လွယ်အောင် ပြောရလျှင် - `update` လုပ်ခြင်းက "သတင်းစာဖတ်ပြီး version အသစ်တွေရှိမရှိ စစ်ဆေးတာ" ဖြစ်ပြီး၊ `upgrade` လုပ်ခြင်းက "အသစ်ထွက်သမျှကို တကယ် install လုပ်လိုက်တာ (ဈေးဝယ်တာ)" နှင့် တူပါသည်။

### 🔄 Update Homebrew
Brew ကိုယ်တိုင်နှင့် package စာရင်းများကို update လုပ်ရန်:
```bash
brew update
```

### ⬆️ Upgrade Packages
Install လုပ်ထားသော software များကို version အသစ်သို့ မြှင့်တင်ရန်:
```bash
brew upgrade
```
