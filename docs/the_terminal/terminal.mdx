import { useState, useEffect } from 'react';

export const ImageLoop = ({ images, width = '100%' }) => {
  const [index, setIndex] = useState(0);
  const [isZoomed, setIsZoomed] = useState(false);
  
  useEffect(() => {
    if (!images || images.length <= 1) return;
    const timer = setInterval(() => {
      setIndex((prev) => (prev + 1) % images.length);
    }, 2000);
    return () => clearInterval(timer);
  }, [images]);

  const containerStyle = {
    cursor: 'zoom-in',
    position: 'relative',
    borderRadius: '12px',
    overflow: 'hidden',
    border: '2px solid var(--ifm-color-emphasis-300)',
    boxShadow: '0 8px 24px rgba(0,0,0,0.15)',
    width: width,
    marginBottom: '20px'
  };

  return (
    <div style={{ marginBottom: '20px' }}>
      <div onClick={() => setIsZoomed(!isZoomed)} style={containerStyle}>
        <img src={images[index]} style={{ display: 'block', width: '100%' }} />
        {images.length > 1 && (
          <div style={{
            position: 'absolute', bottom: '10px', right: '10px',
            backgroundColor: '#333', color: 'white',
            padding: '2px 12px', borderRadius: '20px', fontSize: '0.75rem',
            fontWeight: 'bold', opacity: '0.8'
          }}>
            Step {index + 1} of {images.length}
          </div>
        )}
      </div>
      {isZoomed && (
        <div 
          onClick={() => setIsZoomed(false)}
          style={{
            position: 'fixed', top: 0, left: 0, width: '100%', height: '100%',
            backgroundColor: 'rgba(0,0,0,0.9)', zIndex: 9999,
            display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'zoom-out'
          }}
        >
          <img src={images[index]} style={{ maxHeight: '95vh', maxWidth: '95vw', borderRadius: '8px' }} />
        </div>
      )}
    </div>
  );
};

## Essential Terminal Shortcuts

Command line ကို ကျွမ်းကျင်စွာ အသုံးပြုနိုင်ခြင်းက အချိန်ကုန်သက်သာစေပါသည်။ အသုံးအများဆုံး Shortcut များမှာ အောက်ပါအတိုင်း ဖြစ်ပါသည် -

| Shortcut | Action |
| --- | --- |
| `Ctrl + L` | **Clear screen** (စာမျက်နှာကို ရှင်းထုတ်ရန် - `clear` ဟု ရိုက်ခြင်းနှင့် အတူတူပင်ဖြစ်သည်) |
| `Ctrl + C` | **Abort** (လက်ရှိအလုပ်လုပ်နေသော command သို့မဟုတ် process ကို ရပ်တန့်ရန်) |
| `Ctrl + U` | Cursor မှ အစပိုင်းအထိ တစ်ကြောင်းလုံးကို **ဖျက်ပစ်ရန်** |
| `Ctrl + K` | Cursor မှ အဆုံးပိုင်းအထိ တစ်ကြောင်းလုံးကို **ဖျက်ပစ်ရန်** |
| `Ctrl + A` | Cursor ကို စာကြောင်း၏ **အစ** သို့ ရွှေ့ရန် |
| `Ctrl + E` | Cursor ကို စာကြောင်း၏ **အဆုံး** သို့ ရွှေ့ရန် |
| `Ctrl + D` | လက်ရှိ shell သို့မဟုတ် terminal ကို **ပိတ်ရန်** |

## 🖥️ Command Line Interface (CLI) Basics

**Command Line Interface (CLI)** ဆိုသည်မှာ text-based ပုံစံဖြင့် Operating System ကို ခိုင်းစေနိုင်သော ပတ်ဝန်းကျင်ဖြစ်ပါသည်။ Terminal, Console သို့မဟုတ် Shell ဟု အမျိုးမျိုး ခေါ်ဝေါ်ကြသော်လည်း အခြေခံအကျဆုံး commands များမှာ အတူတူပင် ဖြစ်ပါသည်။

### 🔍 System & Documentation

| Command | Description |
| --- | --- |
| `whoami` | လက်ရှိ login ဝင်ထားသော **User ID** ကို ပြသပေးသည်။ |
| `pwd` | **Print Working Directory**: လက်ရှိရောက်နေသော folder လမ်းကြောင်းကို ပြသပေးသည်။ |
| `man [command]` | Command တစ်ခုချင်းစီ၏ အသုံးပြုပုံလမ်းညွှန် (**Manual**) ကို ဖွင့်ဖတ်နိုင်သည်။ |

### 📂 Navigation & Listing

| Command | Description |
| --- | --- |
| `ls` | Folder ထဲရှိ content များကို **List** အလိုက် ပြသပေးသည်။ |
| `ls -l` | Permissions နှင့် size များအပါအဝင် **long format** ဖြင့် အသေးစိတ်ပြသပေးသည်။ |
| `cd [path]` | သတ်မှတ်ထားသော Folder ထဲသို့ ဝင်ရောက်ရန် (**Change Directory**)။ |
| `cd ..` | လက်ရှိရောက်နေသော folder ၏ **အပေါ်အဆင့် (Parent folder)** သို့ ပြန်ထွက်ရန်။ |
| `cd ../..` | **အပေါ်နှစ်ဆင့်** ပြန်ထွက်ရန်။ |

### 📝 File Manipulation & Editing

| Command | Description |
| --- | --- |
| `mkdir` | **Make Directory**: Folder အသစ်တစ်ခု တည်ဆောက်ရန်။ |
| `mv` | File သို့မဟုတ် Folder များကို **နေရာရွှေ့ရန်** (သို့မဟုတ်) နာမည်ပြောင်းရန်။ |
| `rm` | File များကို **ဖျက်ရန်** (သတိထား၍ အသုံးပြုပါ!)။ |
| `chmod` | **Change Mode**: File တစ်ခု၏ permissions များကို ပြင်ဆင်ရန်။ |
| `cat` | **Concatenate**: File ထဲရှိ အကြောင်းအရာများကို terminal ပေါ်တွင် တိုက်ရိုက်ကြည့်ရှုရန်။ |
| `nano` / `pico` | အသုံးပြုရလွယ်ကူသော **Text Editor** များ။ |
| `vim` | Programmers များအတွက် စွမ်းဆောင်ရည်မြင့်မားသော **Advanced Editor** ဖြစ်သည်။ |

### 🚀 macOS Specific

| Command | Description |
| --- | --- |
| `open .` | လက်ရှိရောက်နေသော Folder ကို **Finder** တွင် ဖွင့်ပေးသည်။ |
| `open -a "App"` | သတ်မှတ်ထားသော **Application** ကို ဖွင့်ပေးသည်။ (ဥပမာ- `open -a "Audio MIDI Setup"`) |

:::danger သတိပြုရန်
