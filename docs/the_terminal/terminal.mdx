import { useState, useEffect } from 'react';

export const ImageLoop = ({ images, width = '100%' }) => {
  const [index, setIndex] = useState(0);
  const [isZoomed, setIsZoomed] = useState(false);
  
  useEffect(() => {
    if (!images || images.length <= 1) return;
    const timer = setInterval(() => {
      setIndex((prev) => (prev + 1) % images.length);
    }, 2000);
    return () => clearInterval(timer);
  }, [images]);

  const containerStyle = {
    cursor: 'zoom-in',
    position: 'relative',
    borderRadius: '12px',
    overflow: 'hidden',
    border: '2px solid var(--ifm-color-emphasis-300)',
    boxShadow: '0 8px 24px rgba(0,0,0,0.15)',
    width: width,
    marginBottom: '20px'
  };

  return (
    <div style={{ marginBottom: '20px' }}>
      <div onClick={() => setIsZoomed(!isZoomed)} style={containerStyle}>
        <img src={images[index]} style={{ display: 'block', width: '100%' }} />
        {images.length > 1 && (
          <div style={{
            position: 'absolute', bottom: '10px', right: '10px',
            backgroundColor: '#333', color: 'white',
            padding: '2px 12px', borderRadius: '20px', fontSize: '0.75rem',
            fontWeight: 'bold', opacity: '0.8'
          }}>
            Step {index + 1} of {images.length}
          </div>
        )}
      </div>
      {isZoomed && (
        <div 
          onClick={() => setIsZoomed(false)}
          style={{
            position: 'fixed', top: 0, left: 0, width: '100%', height: '100%',
            backgroundColor: 'rgba(0,0,0,0.9)', zIndex: 9999,
            display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'zoom-out'
          }}
        >
          <img src={images[index]} style={{ maxHeight: '95vh', maxWidth: '95vw', borderRadius: '8px' }} />
        </div>
      )}
    </div>
  );
};

## Essential Terminal Shortcuts

Managing your command line efficiently can save a massive amount of time. Here are the most common shortcuts for navigation and control:

| Shortcut | Action |
| --- | --- |
| `Ctrl + L` | **Clear screen** (equivalent to typing `clear`) |
| `Ctrl + C` | **Abort** the current command/process |
| `Ctrl + U` | **Delete** the entire line (from cursor to start) |
| `Ctrl + K` | **Delete** from cursor to the end of the line |
| `Ctrl + A` | Move cursor to the **beginning** of the line |
| `Ctrl + E` | Move cursor to the **end** of the line |
| `Ctrl + D` | **Exit** the current shell or close the terminal |

## 🖥️ Command Line Interface (CLI) Basics

A **Command Line Interface (CLI)** is a text-based environment used to interact with an operating system. Whether you call it the terminal, console, or shell, these core commands are your primary tools.

---

### 🔍 System & Documentation

| Command | Description |
| --- | --- |
| `whoami` | Displays the **effective user ID** of the current user. |
| `pwd` | **Print Working Directory**: Shows exactly where you are. |
| `man [command]` | Opens the **manual** (documentation) for a specific command. |

### 📂 Navigation & Listing

| Command | Description |
| --- | --- |
| `ls` | **List** directory contents. |
| `ls -l` | List contents in **long format** (shows permissions, size, etc.). |
| `cd [path]` | **Change Directory**. |
| `cd ..` | Move **up one level** in the directory tree. |
| `cd ../..` | Move **up two levels**. |

### 📝 File Manipulation & Editing

| Command | Description |
| --- | --- |
| `mkdir` | **Make Directory**: Creates a new folder. |
| `mv` | **Move** or rename files and directories. |
| `rm` | **Remove** files or directory entries (use with caution!). |
| `chmod` | **Change Mode**: Modify file permissions or Access Control Lists. |
| `cat` | **Concatenate**: View the contents of a file directly in the terminal. |
| `nano` / `pico` | Simple, user-friendly **text editors**. |
| `vim` | **Vi IMproved**: A powerful, advanced editor for programmers. |

### 🚀 macOS Specific

| Command | Description |
| --- | --- |
| `open .` | Opens the current folder in **Finder**. |
| `open -a "App"` | Opens a specific **application** (e.g., `open -a "Audio MIDI Setup"`). |

---

> [!WARNING]
> The `rm` command is permanent. Unlike moving a file to the Trash, files deleted via CLI are generally not recoverable without backups.
