import { useState, useEffect } from 'react';

export const ImageLoop = ({ images }) => {
  const [index, setIndex] = useState(0);
  const [isZoomed, setIsZoomed] = useState(false);
  
  useEffect(() => {
    if (images.length <= 1) return;
    const timer = setInterval(() => {
      setIndex((prevIndex) => (prevIndex + 1) % images.length);
    }, 2000); // Intro အတွက် နည်းနည်းလေး ပိုနှေးပေးထားပါတယ်
    return () => clearInterval(timer);
  }, [images.length]);

  return (
    <div style={{ marginBottom: '20px' }}>
      <div 
        onClick={() => setIsZoomed(!isZoomed)}
        style={{ 
          cursor: 'zoom-in', 
          position: 'relative',
          borderRadius: '12px',
          overflow: 'hidden',
          border: '2px solid var(--ifm-color-emphasis-300)',
          boxShadow: '0 8px 24px rgba(0,0,0,0.15)'
        }}
      >
        <img src={images[index]} style={{ display: 'block', width: '100%' }} />
        
        {images.length > 1 && (
          <div style={{
            position: 'absolute', bottom: '10px', right: '10px',
            backgroundColor: '#333', color: 'white',
            padding: '2px 12px', borderRadius: '20px', fontSize: '0.75rem', 
            fontWeight: 'bold', opacity: '0.8'
          }}>
            Step {index + 1} of {images.length}
          </div>
        )}
      </div>

      {isZoomed && (
        <div 
          onClick={() => setIsZoomed(false)}
          style={{
            position: 'fixed', top: 0, left: 0, width: '100%', height: '100%',
            backgroundColor: 'rgba(0,0,0,0.9)', zIndex: 9999,
            display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'zoom-out'
          }}
        >
          <img src={images[index]} style={{ maxHeight: '95vh', maxWidth: '95vw', borderRadius: '8px' }} />
        </div>
      )}
    </div>
  );
};

# Introduction to Raspberry Pi Pico 2 W

Raspberry Pi Pico 2 W ဟာ နောက်ဆုံးပေါ် **RP2350** chip ကို အသုံးပြုထားတဲ့ စွမ်းဆောင်ရည်မြင့် microcontroller တစ်ခုဖြစ်ပါတယ်။ Dual-core Processor ပါဝင်ပြီး Wi-Fi (2.4GHz) နဲ့ Bluetooth 5.2 တို့ကိုပါ အပြည့်အဝ Support ပေးထားပါတယ်။



### Key Features
* **Dual-Core Architecture:** တစ်ဖက်က Sensor data တွေကိုဖတ်ပြီး နောက်တစ်ဖက်ကနေ USB signal ပို့တာမျိုး ဒါမှမဟုတ် Wi-Fi ဆက်သွယ်ရေးလုပ်တာမျိုး (Multitasking) ကို အလွယ်တကူ လုပ်ဆောင်နိုင်ပါတယ်။
* **TinyUSB Support:** USB stack ကို အပြည့်အဝ Support ပေးတဲ့အတွက် Keyboard, Mouse သို့မဟုတ် MIDI device တွေအဖြစ် ဖန်တီးဖို့ အဆင်ပြေပါတယ်။
* **Wireless Connectivity:** 2.4GHz Wi-Fi ပါဝင်လို့ Internet of Things (IoT) project တွေအတွက် အသင့်တော်ဆုံးပါပဲ။
* **Security:** Arm TrustZone အခြေခံထားတဲ့ security feature တွေပါဝင်လာလို့ data တွေကို ပိုမိုလုံခြုံစွာ သိမ်းဆည်းနိုင်ပါတယ်။

### သင်ယူသွားမည့် လမ်းညွှန် (Proposed Curriculum)

Pico 2 W က MicroPython ရော C/C++ ရော သုံးလို့ရပါတယ်။ အောက်ပါ သင်ခန်းစာတွေကို အကြံပြုချင်ပါတယ် -

1.  **Lesson 1:** Setting up MicroPython & Thonny IDE (Environment Setup)
2.  **Lesson 2:** Hello LED (Internal & External LED Control)
3.  **Lesson 3:** Reading Inputs (Buttons & Switches)
4.  **Lesson 4:** PWM & ADC (Fading LED & Potentiometer)
5.  **Lesson 5:** Dual Core Programming (Core 0 & Core 1 ခွဲသုံးနည်း)
6.  **Lesson 6:** Wi-Fi Basics (Connecting to Internet)
7.  **Lesson 7:** Simple Web Server (Browser ကနေ LED ထိန်းချုပ်ခြင်း)
