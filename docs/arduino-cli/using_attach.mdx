import { useState, useEffect } from 'react';

export const ImageLoop = ({ images }) => {
  const [index, setIndex] = useState(0);
  const [isZoomed, setIsZoomed] = useState(false);
  
  useEffect(() => {
    if (images.length <= 1) return;
    const timer = setInterval(() => {
      setIndex((prevIndex) => (prevIndex + 1) % images.length);
    }, 2000);
    return () => clearInterval(timer);
  }, [images.length]);

  return (
    <div style={{ marginBottom: '20px' }}>
      <div 
        onClick={() => setIsZoomed(!isZoomed)}
        style={{ 
          cursor: 'zoom-in', 
          position: 'relative',
          borderRadius: '12px',
          overflow: 'hidden',
          border: '2px solid var(--ifm-color-emphasis-300)',
          boxShadow: '0 8px 24px rgba(0,0,0,0.15)'
        }}
      >
        <img src={images[index]} style={{ display: 'block', width: '100%' }} />
        
        {images.length > 1 && (
          <div style={{
            position: 'absolute', bottom: '10px', right: '10px',
            backgroundColor: '#333', color: 'white',
            padding: '2px 12px', borderRadius: '20px', fontSize: '0.75rem', 
            fontWeight: 'bold', opacity: '0.8'
          }}>
            Step {index + 1} of {images.length}
          </div>
        )}
      </div>

      {isZoomed && (
        <div 
          onClick={() => setIsZoomed(false)}
          style={{
            position: 'fixed', top: 0, left: 0, width: '100%', height: '100%',
            backgroundColor: 'rgba(0,0,0,0.9)', zIndex: 9999,
            display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'zoom-out'
          }}
        >
          <img src={images[index]} style={{ maxHeight: '95vh', maxWidth: '95vw', borderRadius: '8px' }} />
        </div>
      )}
    </div>
  );
};

# Using Attach

`board attach` command သည် သင်၏ sketch နှင့် board configuration (Core & Port) ကို တစ်ခါတည်း တွဲပေးထားခြင်းဖြစ်ပါသည်။ ဤသို့လုပ်ဆောင်ထားခြင်းဖြင့် နောက်ပိုင်းတွင် command အရှည်ကြီးများ ရိုက်စရာမလိုဘဲ compile နှင့် upload ကို အလွယ်တကူ ပြုလုပ်နိုင်မည် ဖြစ်ပါသည်။

### 1. Board နှင့် Sketch ကို ချိတ်ဆက်ခြင်း (Attach)
ပထမဦးစွာ မိမိအသုံးပြုမည့် board ၏ core နှင့် port ကို အောက်ပါအတိုင်း တစ်ကြိမ် attach လုပ်ပေးရပါမည်။

```bash
arduino-cli board attach -b arduino:avr:uno -p /dev/cu.usbmodem1101 MySketch
```

### 2. Configuration File ကို စစ်ဆေးခြင်း
Attach လုပ်ပြီးပါက `MySketch` folder ထဲတွင် `sketch.yaml` ဆိုသော file လေး ရောက်ရှိလာပါမည်။ ၎င်း file ထဲတွင် configuration များကို အောက်ပါအတိုင်း စစ်ဆေးနိုင်ပါသည်။

```bash
ls MySketch
cat MySketch/sketch.yaml
```

### 3. ရိုးရှင်းစွာ Compile နှင့် Upload လုပ်ခြင်း
တစ်ကြိမ် attach လုပ်ပြီးသွားလျှင် port နှင့် board name များ ထပ်ထည့်ရန်မလိုတော့ဘဲ compile နှင့် upload လုပ်နိုင်ပါသည်။

**Compile လုပ်ရန်:**
```bash
arduino-cli compile MySketch
```

### 4. Sketch Folder ထဲသို့ဝင်၍ အသုံးပြုခြင်း
အကယ်၍ သင်သည် sketch folder ထဲသို့ `cd` ဖြင့် ဝင်ရောက်ထားပါက ပို၍ပင် တိုတောင်းသော command များကို အသုံးပြုနိုင်ပါသည်။

```bash
cd MySketch

# Compile လုပ်ရန်
arduino-cli compile

# Upload လုပ်ရန်
arduino-cli upload
```
