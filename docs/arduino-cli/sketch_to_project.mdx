import { useState, useEffect } from 'react';

export const ImageLoop = ({ images }) => {
  const [index, setIndex] = useState(0);
  const [isZoomed, setIsZoomed] = useState(false);
  
  useEffect(() => {
    if (images.length <= 1) return;
    const timer = setInterval(() => {
      setIndex((prevIndex) => (prevIndex + 1) % images.length);
    }, 2000);
    return () => clearInterval(timer);
  }, [images.length]);

  return (
    <div style={{ marginBottom: '20px' }}>
      <div 
        onClick={() => setIsZoomed(!isZoomed)}
        style={{ 
          cursor: 'zoom-in', 
          position: 'relative',
          borderRadius: '12px',
          overflow: 'hidden',
          border: '2px solid var(--ifm-color-emphasis-300)',
          boxShadow: '0 8px 24px rgba(0,0,0,0.15)'
        }}
      >
        <img src={images[index]} style={{ display: 'block', width: '100%' }} />
        
        {images.length > 1 && (
          <div style={{
            position: 'absolute', bottom: '10px', right: '10px',
            backgroundColor: '#333', color: 'white',
            padding: '2px 12px', borderRadius: '20px', fontSize: '0.75rem', 
            fontWeight: 'bold', opacity: '0.8'
          }}>
            Step {index + 1} of {images.length}
          </div>
        )}
      </div>

      {isZoomed && (
        <div 
          onClick={() => setIsZoomed(false)}
          style={{
            position: 'fixed', top: 0, left: 0, width: '100%', height: '100%',
            backgroundColor: 'rgba(0,0,0,0.9)', zIndex: 9999,
            display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'zoom-out'
          }}
        >
          <img src={images[index]} style={{ maxHeight: '95vh', maxWidth: '95vw', borderRadius: '8px' }} />
        </div>
      )}
    </div>
  );
};

# Sketch to Project

Arduino ပရောဂျက်အသစ်တစ်ခုကို Terminal မှတစ်ဆင့် တည်ဆောက်ပုံ အဆင့်ဆင့် ဖြစ်ပါသည်။

```bash
ls
```

### 4. Code ရေးသားခြင်း (Vim Editor အသုံးပြုနည်း)
`.ino` file ထဲသို့ code များရေးသားရန် Vim editor ကို အသုံးပြုပါမည်။

```bash
vim MySketch/MySketch.ino
```

**Vim အသုံးပြုပုံ အခြေခံအဆင့်ဆင့်:**
1. **Insert Mode:** စာစရေးရန် `i` သို့မဟုတ် `a` ကို နှိပ်ပါ။
2. **Writing Code:** အောက်ပါ code ကို ရိုက်ထည့်ပါ။

```cpp
void setup() {
  Serial.begin(9600);
  while (!Serial);
}

void loop() {
  Serial.println("Hello, World!");
  delay(1000);
}
```

3. **Save & Exit:** 
  * `Esc` ကို နှိပ်ပါ။
  * `:wq` (Write and Quit) ကို ရိုက်ပြီး `Enter` ခေါက်ပါ။

### 5. ရေးသားထားသော Code ကို ပြန်လည်ကြည့်ရှုခြင်း
File ထဲသို့ code များ မှန်ကန်စွာ ရောက်မရောက်ကို `cat` command ဖြင့် စစ်ဆေးနိုင်ပါသည်။

```bash
cat MySketch/MySketch.ino
```
