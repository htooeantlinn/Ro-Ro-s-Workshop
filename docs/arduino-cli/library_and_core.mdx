import { useState, useEffect } from 'react';

export const ImageLoop = ({ images }) => {
  const [index, setIndex] = useState(0);
  const [isZoomed, setIsZoomed] = useState(false);
  
  useEffect(() => {
    if (images.length <= 1) return;
    const timer = setInterval(() => {
      setIndex((prevIndex) => (prevIndex + 1) % images.length);
    }, 2000);
    return () => clearInterval(timer);
  }, [images.length]);

  return (
    <div style={{ marginBottom: '20px' }}>
      <div 
        onClick={() => setIsZoomed(!isZoomed)}
        style={{ 
          cursor: 'zoom-in', 
          position: 'relative',
          borderRadius: '12px',
          overflow: 'hidden',
          border: '2px solid var(--ifm-color-emphasis-300)',
          boxShadow: '0 8px 24px rgba(0,0,0,0.15)'
        }}
      >
        <img src={images[index]} style={{ display: 'block', width: '100%' }} />
        
        {images.length > 1 && (
          <div style={{
            position: 'absolute', bottom: '10px', right: '10px',
            backgroundColor: '#333', color: 'white',
            padding: '2px 12px', borderRadius: '20px', fontSize: '0.75rem', 
            fontWeight: 'bold', opacity: '0.8'
          }}>
            Step {index + 1} of {images.length}
          </div>
        )}
      </div>

      {isZoomed && (
        <div 
          onClick={() => setIsZoomed(false)}
          style={{
            position: 'fixed', top: 0, left: 0, width: '100%', height: '100%',
            backgroundColor: 'rgba(0,0,0,0.9)', zIndex: 9999,
            display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'zoom-out'
          }}
        >
          <img src={images[index]} style={{ maxHeight: '95vh', maxWidth: '95vw', borderRadius: '8px' }} />
        </div>
      )}
    </div>
  );
};

# Library & Core Management

Arduino ပရောဂျက်များတွင် လိုအပ်သော Library များနှင့် Board Core (Platforms) များကို Terminal မှ စီမံခန့်ခွဲနည်းများ ဖြစ်ပါသည်။

## 📚 Library စီမံခန့်ခွဲခြင်း

### Library အသစ်ထည့်သွင်းခြင်း
ဥပမာအနေဖြင့် MIDI Library ကို အောက်ပါအတိုင်း install လုပ်နိုင်ပါသည်။
```bash
arduino-cli lib install "MIDI Library"
```

## 🔌 Specific Board Cores Installation

### 1. Arduino UNO (AVR)
```bash
arduino-cli core install arduino:avr
```

### 2. ESP32 Series
အင်တာနက်လိုင်းကြောင့် error တက်ခြင်းမှ ကာကွယ်ရန် retry loop ဖြင့် ထည့်သွင်းနည်း ဖြစ်ပါသည်။
```bash
export ARDUINO_CLI_DOWNLOAD_TIMEOUT=600; until arduino-cli core install esp32:esp32; do echo "Retrying in 15s..."; sleep 15; done
```

### 3. Raspberry Pi Pico 2W (RP2040)
ပထမဦးစွာ URL ထည့်သွင်းပြီးမှ install ပြုလုပ်ရပါမည်။

```bash
# Additional URL ထည့်သွင်းခြင်း
arduino-cli config add board_manager.additional_urls [https://github.com/earlephilhower/arduino-pico/releases/download/global/package_rp2040_index.json](https://github.com/earlephilhower/arduino-pico/releases/download/global/package_rp2040_index.json) && arduino-cli core update-index

# Core install လုပ်ခြင်း
export ARDUINO_CLI_DOWNLOAD_TIMEOUT=1200; until arduino-cli core install rp2040:rp2040; do echo "Retrying in 10s..."; sleep 10; done
```
:::info Pro-Tip
Pico အတွက် `picotool` ကိုပါ တစ်ခါတည်း install လုပ်ထားသင့်ပါသည်။
```bash
brew install picotool
```
:::
